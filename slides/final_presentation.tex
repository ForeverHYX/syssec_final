\section{背景与目标}

\begin{frame}{背景问题}{加固技术隐藏什么，检测器要回答什么}
  \centering
  \begin{tikzpicture}[node distance=0.56cm]
    \node[hiBoxOrange, minimum width=3.35cm, minimum height=2.65cm] (hardening) {
      \textbf{加固让分析变难}\\[0.35em]
      \scriptsize 壳 stub 接管入口\\
      \scriptsize DEX 或 payload 动态加载\\
      \scriptsize 高熵资源隐藏真实代码
    };
    \node[hiBoxGreen, right=of hardening, minimum width=3.45cm, minimum height=2.65cm] (evidence) {
      \textbf{静态可见证据}\\[0.35em]
      \scriptsize Manifest 入口类\\
      \scriptsize DEX 字符串与方法表\\
      \scriptsize Native 字符串与 JNI 入口
    };
    \node[hiBox, right=of evidence, minimum width=3.45cm, minimum height=2.65cm] (output) {
      \textbf{项目输出}\\[0.35em]
      \scriptsize packer / obfuscation\\
      \scriptsize environment / native\\
      \scriptsize JSON 证据链与统计指标
    };
    \draw[hiArrow] (hardening) -- (evidence);
    \draw[hiArrow] (evidence) -- (output);
  \end{tikzpicture}

  \vspace{0.8em}
  HardenInspector 的目标不是“判恶意”，而是回答：一个 APK 使用了哪些加固/反分析技术，证据在哪里。
\end{frame}

\begin{frame}{项目目标}{检测器，而不是通用反编译器或恶意判定器}
  \begin{columns}[T,totalwidth=\textwidth]
    \begin{column}{0.48\textwidth}
      \begin{block}{输入}
        Android APK：Manifest、DEX、Native 库、资源与 assets。
      \end{block}
      \begin{block}{输出}
        四类技术画像：加壳、混淆、环境检测、Native 执行路径。
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{证据}
        每条 finding 给出位置、命中值、类别、置信度和解释。
      \end{block}
      \begin{block}{边界}
        不判断恶意/良性；不复制开源工具规则；动态 Frida 验证作为后续扩展。
      \end{block}
    \end{column}
  \end{columns}
  \vspace{0.5em}
  现实调整：真实商业壳样本获取和分发不稳定，最终交付可复现合成 APK 数据集，并在文档中记录替代关系。
\end{frame}

\begin{frame}{成果一页看懂}
  \begin{columns}[T,totalwidth=\textwidth]
    \begin{column}{0.48\textwidth}
      \begin{block}{我们实现了什么}
        \begin{itemize}
          \item 一个可运行 CLI：输入 APK，输出 JSON/text report。
          \item 一个可复现数据集：11 个 APK、labels 和逐样本报告。
          \item 一个公平 benchmark：只统计本仓库可运行的对比工具。
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{听众应该带走什么}
        \begin{itemize}
          \item 加固检测不是判恶意，而是定位证据。
          \item 证据来自 Manifest、DEX、Native 和资源四类结构。
          \item 输出可以复核：每条 finding 都有 location/value。
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
  \vspace{0.45em}
  所有源码、数据、报告、slides 和开箱环境都维护在同一个仓库中。
\end{frame}

\section{检测方法}

\begin{frame}{APK 里面看什么}{复杂对象先拆成四类证据源}
  \begin{columns}[T,totalwidth=\textwidth]
    \begin{column}{0.56\textwidth}
      \centering
      \includegraphics[width=\textwidth,height=0.62\textheight,keepaspectratio]{apk_static_analysis_cutaway.png}
    \end{column}
    \begin{column}{0.42\textwidth}
      \begin{block}{四类解析对象}
        \begin{itemize}
          \item Manifest：入口类、权限、硬编码字符串。
          \item DEX：类名、方法名、字符串池和调用痕迹。
          \item Native：\texttt{JNI\_OnLoad}、反调试和 Hook 字符串。
          \item assets/resources：高熵 payload 和可疑文件路径。
        \end{itemize}
      \end{block}
      \vspace{0.3em}
      这张图只承担“APK 拆解对象”的直观说明；具体规则和指标由后续表格给出。
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{总体架构}{从 APK 到 evidence-chain 报告}
  \centering
  \begin{tikzpicture}[node distance=0.42cm]
    \node[hiBox, minimum width=2.55cm, minimum height=3.05cm] (input) {
      \textbf{输入 APK}\\[0.35em]
      \scriptsize ZIP entries\\
      \scriptsize Manifest\\
      \scriptsize classes.dex\\
      \scriptsize lib/*.so / assets
    };
    \node[hiBoxGreen, right=of input, minimum width=3.0cm, minimum height=3.05cm] (parse) {
      \textbf{解析层}\\[0.35em]
      \scriptsize 文件清单与熵值\\
      \scriptsize AXML string pool\\
      \scriptsize DEX strings/types/methods\\
      \scriptsize Native printable strings
    };
    \node[hiBoxOrange, right=of parse, minimum width=3.0cm, minimum height=3.05cm] (feature) {
      \textbf{特征层}\\[0.35em]
      \scriptsize 壳库名 / StubApp\\
      \scriptsize 动态加载 API\\
      \scriptsize 短类名比例 / 反射\\
      \scriptsize 模拟器 / 调试 / Hook
    };
    \node[hiBox, right=of feature, minimum width=2.75cm, minimum height=3.05cm] (report) {
      \textbf{规则与报告}\\[0.35em]
      \scriptsize 解释型规则组合\\
      \scriptsize confidence / severity\\
      \scriptsize evidence location/value\\
      \scriptsize JSON + text output
    };
    \draw[hiArrow] (input) -- (parse);
    \draw[hiArrow] (parse) -- (feature);
    \draw[hiArrow] (feature) -- (report);
  \end{tikzpicture}

  \vspace{0.7em}
  技术路线遵照中期报告：核心检测器使用自写轻量静态解析器；APKiD、Androguard、DroidLysis、MobSF 只作为 benchmark 或定性参照。
\end{frame}

\begin{frame}{证据链怎么生成}
  \begin{columns}[T,totalwidth=\textwidth]
    \begin{column}{0.46\textwidth}
      \begin{block}{解析得到原始事实}
        \begin{itemize}
          \item APK：文件清单、压缩大小、熵值、资源路径。
          \item Manifest：壳入口类、权限、硬编码字符串。
          \item DEX：字符串池、类型描述符、方法名、\texttt{const-string}、\texttt{invoke-*}。
          \item Native：\texttt{JNI\_OnLoad}、反调试和 Hook 相关字符串。
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.52\textwidth}
      \begin{block}{组合成可复核 finding}
        \scriptsize
        \begin{tabular}{p{1.55cm}p{4.35cm}}
          \toprule
          字段 & 示例 \\
          \midrule
          category & \texttt{packer} \\
          rule & \texttt{packer.dynamic\_code\_loading} \\
          evidence & \texttt{classes.dex:string}\\
                   & \texttt{= DexClassLoader} \\
          confidence & \texttt{medium} \\
          explanation & 动态加载 API 可用于壳或插件加载 \\
          \bottomrule
        \end{tabular}
      \end{block}
      \vspace{0.35em}
      这样听众可以知道“为什么判定”，而不是只看到一个不可解释标签。
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{检测规则示例}
  \scriptsize
  \begin{table}
    \centering
    \begin{tabular}{p{1.4cm}p{6.5cm}p{2.4cm}}
      \toprule
      类别 & 规则信号 & 输出证据 \\
      \midrule
      加壳 & 壳库名、StubApp、高熵 payload、动态加载 API & packer finding \\
      混淆 & 短类名比例、反射 API、字符串/调用组合 & obfuscation finding \\
      环境检测 & emulator properties、debugger、Frida/Xposed & environment finding \\
      Native & JNI 入口、native loader、反分析字符串 & native finding \\
      \bottomrule
    \end{tabular}
  \end{table}
  \vspace{0.4em}
  每条 finding 都包含 category、confidence、description 和 evidence，便于人工复核。
\end{frame}

\begin{frame}{控制流轻量统计}{解决“只统计 opcode 但未成规则”的局限}
  \centering
  \begin{tikzpicture}[node distance=0.42cm]
    \node[hiBoxOrange, minimum width=3.0cm, minimum height=2.8cm] (opcode) {
      \textbf{DEX code item}\\[0.35em]
      \scriptsize \texttt{if-*} 条件分支\\
      \scriptsize \texttt{goto} 跳转\\
      \scriptsize \texttt{packed/sparse-switch}\\
      \scriptsize \texttt{throw}
    };
    \node[hiBoxGreen, right=of opcode, minimum width=3.25cm, minimum height=2.8cm] (profile) {
      \textbf{Opcode profile}\\[0.35em]
      \scriptsize instruction count\\
      \scriptsize control-flow count\\
      \scriptsize branch density\\
      \scriptsize if/goto/switch 分项
    };
    \node[hiBox, right=of profile, minimum width=3.35cm, minimum height=2.8cm] (finding) {
      \textbf{Finding}\\[0.35em]
      \scriptsize \texttt{obfuscation.}\\
      \scriptsize \texttt{control\_flow\_density}\\
      \scriptsize evidence: \texttt{classes.dex}\\
      \scriptsize density 与计数明细
    };
    \draw[hiArrow] (opcode) -- (profile);
    \draw[hiArrow] (profile) -- (finding);
  \end{tikzpicture}

  \vspace{0.5em}
  这不是完整 CFG；它是可复现、低依赖、可解释的预筛信号，用于提示“这里可能存在控制流平坦化或跳转扰动”。
\end{frame}

\begin{frame}{外部 APK 语料}{不只扫描自构造样本}
  \small
  \begin{columns}[T,totalwidth=\textwidth]
    \begin{column}{0.42\textwidth}
      \begin{block}{来源组成}
        \begin{itemize}
          \item DroidBench：10 个现成测试 APK。
          \item F-Droid：1 个真实开源 APK。
          \item PIVAA：1 个漏洞测试 APK。
        \end{itemize}
      \end{block}
      \vspace{0.35em}
      外部 APK 无 hardening oracle，不计算 F1；只统计覆盖率和 finding 分布。

    \begin{column}{0.56\textwidth}

          APKiD & 12/12 & 2 & 0 & 0 \\
          Androguard DEX & 12/12 & 8 & 3 & 6 \\
          ZIP Strings & 12/12 & 9 & 3 & 6 \\
          \bottomrule
        \end{tabular}
      \end{table}
      \vspace{0.25em}
      \scriptsize F-Droid 样本当前无 finding；外部语料帮助发现并收紧了过敏的控制流密度规则。

\begin{frame}{Benchmark 公平口径}
  \scriptsize
  \begin{table}
    \centering
    \begin{tabular}{p{2.25cm}p{1.55cm}p{7.55cm}}
      \toprule
      对象 & 评分？ & 原因 \\
      \midrule
      HardenInspector & 是 & 本项目检测器，四类 evidence-chain 输出 \\
      APKiD & 是 & 可安装可运行，输出 packer/oddity 相关签名 \\

  \end{table}
  评分只比较当前仓库环境中 11/11 样本可运行的工具，避免把“环境不可用”统计成工具能力差。

\section{交付与结论}

\begin{frame}{开箱即用环境}
  \begin{columns}[T,totalwidth=\textwidth]
    \begin{column}{0.48\textwidth}
      \begin{block}{本地环境}
        \texttt{make setup}\\
        \texttt{make test}\\
        \texttt{make dataset}\\
        \texttt{make benchmark}\\
        \texttt{make demo}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{容器环境}
        \texttt{docker build -t hardeninspector .}\\
        \texttt{docker run --rm hardeninspector make test}
      \end{block}
    \end{column}
  \end{columns}
  \vspace{0.7em}
  Fresh venv 验证：测试通过，CLI help、数据集生成、benchmark 和 slides 编译均可复现。
\end{frame}

\begin{frame}{项目交付}
  \begin{itemize}
    \item 检测器源码：\texttt{src/hardeninspector/}
    \item 数据集：\texttt{datasets/hardeninspector\_eval\_v1/}
    \item 外部 APK 语料：\texttt{datasets/external\_apk\_corpus\_v1/}
    \item Benchmark 统计：\texttt{reports/benchmark/}
    \item 外部 APK 统计：\texttt{reports/external\_corpus/}
    \item 中文文档：\texttt{README.md}、\texttt{docs/}
    \item 总结报告：\texttt{reports/final\_summary.md}
    \item ZJU Beamer：\texttt{slides/final\_presentation.tex}
  \end{itemize}
\end{frame}

\begin{frame}{局限与已优化项}
  \begin{columns}[T,totalwidth=\textwidth]
    \begin{column}{0.48\textwidth}
      \begin{block}{这轮已补齐}
        \begin{itemize}
          \item opcode 统计升级为 branch/goto/switch/throw profile。
          \item 增加 \texttt{obfuscation.control\_flow\_density} 规则。
          \item 数据集加入 \texttt{control\_flow\_flattening} 样本。
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{仍然保留边界}
        \begin{itemize}
          \item 不宣称完整 CFG 或深度 Native 反汇编。
          \item 不把动态 Frida 验证作为必需交付。
          \item 合成数据集不等价于真实大规模 APK 生态。
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{结论}
  \begin{itemize}
    \item HardenInspector 完成了中期报告提出的轻量静态证据链检测器。
    \item 构造了 11 个样本的可复现数据集，并给出 Precision、Recall、Micro F1、Macro F1 统计。
    \item 与 APKiD 等开源工具对比后，当前实现已覆盖课程目标范围内的关键信号。
    \item 工具、数据集、文档、环境和汇报材料均已维护在 GitHub 仓库中。
  \end{itemize}
\end{frame}
